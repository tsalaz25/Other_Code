#include <stdio.h>
char board[9][9];
char sudokuIn[100];
int boardFull = 0;
int isLine = 1;

void initBoard (){
  printf("in initBoard\n");
  int row,col,i,j,k,numchars;
  i = numchars = 0;
  if (fgets(sudokuIn, sizeof(sudokuIn), stdin)){
    for(j = 0; j < 81; j++){
      ((sudokuIn[j] > 47 && sudokuIn[j] < 58) || sudokuIn[j] == 46) ?
        (numchars++) : (boardFull = 0) ;
    }
  } else {
    isLine = 0;
  }
  /*Echo For Loop*/
  for (k = 0; k < sizeof(sudokuIn); k++){
    printf("%c" , sudokuIn[k]);
  }
  (numchars == 81) ? (boardFull = 1) : (printf("error\n"));
  /*(numEmpty + numSolved == 81) ? printf("True") : printf("falase") ;*/

  for (row = 0; row < 9; row++){
    for (col = 0; col < 9; col++){
      board[row][col] = sudokuIn[i];
      i++;
    }
  }
}

void printBoard(){
  int row, col;
  for (row = 0; row < 9; row++){
    for (col = 0; col < 9; col++){
      printf ("%c ",board[row][col]);
    }
    printf("\n");
  }
}

int checkRow(char sudoku[9][9], int row, int n){
  int i;
  for (i = 0; i < 9; i++){
    if (sudoku[row][i] == n){
      return 1;
    }
  }
  return 0;
}

int checkCol(char sudoku[9][9], int col, int n){
  int i;
  for (i = 0; i < 9; i++){
    if (sudoku[i][col] == n){
      return 1;
    }
  }
  return 0;
}

int checkBox(char sudoku[9][9], int row, int col, int n){
  int i,j;
  int boxRow = row - row % 3;
  int boxCol = col - col % 3;

  for (i = boxRow; i < boxRow + 3; i++){
    for (j = boxCol; i < boxCol + 3; j++){
      if (sudoku[i][j] == n){
        return 1;
      }
    }
  }
  return 0;
}

int validCell(char sudoku[9][9], int row, int col, int n){
  return (checkRow(sudoku,row,n) && checkCol(sudoku,col,n) &&
          checkBox(sudoku,row,col,n));
}

void resetBoard(){
  int row, col;
  int i = 0;
  boardFull = 0;
  for (row = 0; row < 9; row++){
    for (col = 0; col < 9; col++){
      board[row][col] = 46;
    }
  }
  while (i < 100){
    sudokuIn[i] = 0;
  }
}

void boardSolved (char solved[9][9]){
  int row, col;
  for (row = 0; row < 9; row ++){
    for (col = 0; col < 9; col++){
      printf ("%c" , solved[row][col]);
    }
  }
  printf("\n");
  resetBoard();
}

int solve(char sudoku[9][9]){
  printf("in solve funtion");
  int row, col, n;
  for (row = 0; row < 9; row++){
    for (col = 0; col < 9; col++){
      if (sudoku[row][col] == 46){
        for (n = 1; n <= 9; n++){
          if (validCell(board,row,col,n)){
            sudoku[row][col] = (char)n;

            if(solve(sudoku)){
              boardSolved(sudoku);
              return 1;
            }else{
              sudoku[row][col] = '.';
            }
          }
        }
        resetBoard();
        printf("\n");
        return 0;
      }
    }
  }
  boardSolved(sudoku);
  return 1;
}

int main (){
  if (isLine == 1){
    initBoard();
    printBoard();
    solve(board);
    printBoard();
  }
  return 0;
}
