#include <stdio.h>
char board[9][9];
char sudokuIn[100];

void printBoard(){
  int row, col;
  for (row = 0; row < 9; row++){
    for (col = 0; col < 9; col++){
      printf ("%c ",board[row][col]);
    }
    printf("\n");
  }
}

int checkRow(char sudoku[9][9], int row, int n){
  int i;
  for (i = 0; i < 9; i++){
    if (sudoku[row][i] == n){
      return 1;
    }
  }
  return 0;
}

int checkCol(char sudoku[9][9], int col, int n){
  int i;
  for (i = 0; i < 9; i++){
    if (sudoku[i][col] == n){
      return 1;
    }
  }
  return 0;
}

int checkBox(char sudoku[9][9], int row, int col, int n){
  int i,j;
  int boxRow = row - row % 3;
  int boxCol = col - col % 3;

  for (i = boxRow; i < boxRow + 3; i++){
    for (j = boxCol; j < boxCol + 3; j++){
      if (sudoku[i][j] == n){
        return 1;
      }
    }
  }
  return 0;
}

int validCell(char sudoku[9][9], int row, int col, int n){
  return (checkRow(sudoku,row,n) && checkCol(sudoku,col,n) &&
          checkBox(sudoku,row,col,n));
}
